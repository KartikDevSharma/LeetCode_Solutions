```Python
class Solution:
    def generateKey(self, num1: int, num2: int, num3: int) -> int:
        def get_digit(n, pos):
            return (n // (10 ** pos)) % 10

        key = 0
        for i in range(3, -1, -1):
            min_digit = min(get_digit(num1, i), get_digit(num2, i), get_digit(num3, i))
            key = key * 10 + min_digit

        return key
```
