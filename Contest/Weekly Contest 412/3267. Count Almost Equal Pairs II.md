class Solution {
		public int countPairs(int[] nums) {
			List<Set<Integer>> ls = new ArrayList<>();
			int n = nums.length;
			for(int i = 0;i < n;i++){
				ls.add(new HashSet<>());
			}
			Map<Integer, long[]> map = new HashMap<>();
			for(int i = 0;i < n;i++){
				int v = nums[i];
				char[] s = String.format("%07d", v).toCharArray();

				if(!map.containsKey(v)){
					map.put(v, new long[80]);
				}
				map.get(v)[i>>>6] |= 1L << i;

				ls.get(i).add(v);
				for(int j = 0;j < 7;j++){
					for(int k = j+1;k < 7;k++){
						{char d = s[j];s[j] = s[k];s[k] = d;}
						int w = 0;
						for(char c : s)w = w * 10 + c-'0';
						ls.get(i).add(w);

						if(!map.containsKey(w)){
							map.put(w, new long[80]);
						}
						map.get(w)[i>>>6] |= 1L << i;

						{char d = s[j];s[j] = s[k];s[k] = d;}
					}
				}
			}

			int ret = 0;
			for(int i = 0;i < n;i++){
				long[] u = new long[80];
				for(int v : ls.get(i)) {
					for (int j = 0; j < 80; j++) {
						u[j] |= map.get(v)[j];
					}
				}
				int num = 0;
				for(long v : u)num += Long.bitCount(v);
				ret += num - 1;
			}
			return ret / 2;
		}
	}
